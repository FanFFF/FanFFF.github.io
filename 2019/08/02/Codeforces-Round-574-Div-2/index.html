A. Drinks Choosing

有n个学生，k种饮料，a1~ai是第i个学生最喜欢的饮料，可以订n/2箱饮料，一箱两瓶，一人一瓶，如果多出来饮料则由老师喝掉，问学生喝到自己喜欢饮料的最大人数是多少？
思路：记录喜欢每个口味的人数，然后以2为基数往下减，最后剩下一堆1人的情况，满足一半即可。

#include "bits/stdc++.h"
using namespace std;
int a[1005];
int mp[1005];
int main()
{
    int n, k;
    cin >> n >> k;
    for (int i = 0; i < n; ++i)
    {
        scanf("%d", &a[i]);
        mp[a[i]]++;
    }
    int cnt = 0;
    int ans = 0;
    for (int i = 1; i <= k; i++)
    {
        ans += mp[i] / 2;
        cnt += mp[i] % 2;
    }
    ans *= 2;
    ans += (cnt + 1) / 2;
    cout << ans << endl;

}

B. Sport Mafia

要求用恰好n次操作使最终糖果数为k，操作规则：
1.加入数量为上一次加入糖果+1的糖果。
2.吃掉一个糖果
问一共吃掉多少糖果？
思路：假设放了 $m$次糖果，那么吃糖果的次数q为 $q=n-m$ 。
则有公式：$k+q=((m+1)\times m)/2$
化简得：$m\times (m+3)=2\times(n+k)$
复杂度$O(\sqrt[2]{n+k})$ 可直接搜索

#include "bits/stdc++.h"
using namespace std;
int a[1005];
int main()
{
    long long int n,k;
    scanf("%lld %lld",&n,&k);
    long long int ans = 0;
    for(long long int i = 1; i*i <= (n+k)*2; i++)
    {
        if(i*(i+3)==(n+k)*2)
        {
            ans = n - i;
            break;
        }
    }
    printf("%lld\n",ans);
}

C. Basketball Exercise

有两排人每排n个，从左到右为1 - n, 然后从中选出任意个人，但是又一定的规则
1.连续的两个人不能再同一行
2.下一个人的下标一定要比前一个大
问能选出来的人的身高最多又多大？
思路：典型的动态规划，线性跑一遍维护最大值即可
状态转移方程：
$dp[i][0] = max(dp[i - 1][0], dp[i - 1][1] + a[i])$
$dp[i][1] = max(dp[i - 1][1], dp[i - 1][0] + b[i])$

#include "bits/stdc++.h"
using namespace std;
long long int a[100005];
long long int b[100005];
long long int dp[100005][2];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i = 1; i <= n; i++)
    {
        scanf("%lld",&a[i]);
    }
    for(int i = 1; i <= n; i++)
    {
        scanf("%lld",&b[i]);
    }
    dp[1][0] = a[1];
    dp[1][1] = b[1];
    for(int i = 2; i <= n; i++)
    {
        dp[i][0] = max(dp[i-1][0],dp[i-1][1]+a[i]);
        dp[i][1] = max(dp[i-1][1],dp[i-1][0]+b[i]);
    }
    long long int ans = max(dp[n][0],dp[n][1]);
    printf("%lld\n",ans);
}

D1. Submarine in the Rybinsk Sea (easy edition)

定义拼接函数
$f(a_1a_2…a_p,b_1b_2…b_q):$
$f(a_1…ap,b1…b_q)=\lbrace a_1a_2…a_{p-q+1}b_1a_{p-q+2}b_2…a_{p-1}b_{q-1}a_qb_p\rbrace \tag{p≤q}$
$f(a_1…ap,b1…b_q)=\lbrace b_1b_2…b_{p-q}a_1b_{p-q+1}a_2…a_{p-1}b_{q-1}a_pb_q\rbrace \tag{p<q}$
思路：ai位数相同，分别计算贡献即可。

#include "bits/stdc++.h"

using namespace std;
const int mod = 998244353;
long long a[100004];

int main() {
    long long n;
    cin >> n;
    long long ans = 0LL;
    for (int i = 1; i <= n; ++i) {
        scanf("%lld", &a[i]);
        long long temp = a[i];
        long long base = 1LL;
        while (temp) {
            ans += base * (temp % 10) % mod;
            ans %= mod;
            base *= 100LL;
            base %= mod;
            temp /= 10;
        }
        temp = a[i];
        base = 10;
        while (temp) {
            ans += base * (temp % 10) % mod;
            ans %= mod;
            base *= 100LL;
            base %= mod;
            temp /= 10;
        }
    }
    cout << ans * n % mod << endl;
}

E. OpenStreetMap

给了一种矩阵构造方式，在构造完后的矩阵中，要求所有大小为$a\times 0$的子矩阵的最小值的和。
思路：好像很容易想到二维RMQ，但是我忘了咋写了(后来验证空间不够开，是种错误想法)，于是寻找另一种方案，发现$a\times b$的子矩阵是固定大小，那么从基础矩阵出发，可以进行如下操作：
例如，对于n=5 m=5 a=2 b=2
矩阵为：

1 5 13 29 2
7 17 37 18 39
22 47 38 20 43
30 4 11 25 53
50 44 32 8 19

子矩阵为$2\times 2$，对于每行数据，2个为一组，取最小的，得到如下矩阵：

1  5  13 2
7  17 18 18
22 38 20 20
4  4  11 25
44 32 8  8

在对于每列数据，2个为一组，同上操作：

1 5  13 2
7 17 18 18
4 4  11 20
4 4  8  8

如此我们得到了一个$n-a+1\times m-b+1$的矩阵，加和即可，每一位代表其子矩阵的最小值，其中使用双端队列维护区间最小值。

#include "bits/stdc++.h"
typedef long long LL;
using namespace std;
const int mod = 998244353;
int n, m, a, b;
LL g, x, y, z;
int mp[3002][3002], dp[3002][3002];
deque<pair<int, int> > q;

int main() {
    scanf("%d%d%d%d", &n, &m, &a, &b);;
    scanf("%lld%lld%lld%lld", &g, &x, &y, &z);
    for(int i = 1; i <= n; ++i) {
        for(int j = 1; j <= m; ++j) {
            mp[i][j] = g;
            g = (1LL * g * x % z + y) % z;
        }
    }
    for(int i = 1; i <= n; ++i) {
        while(!q.empty()) q.pop_back();
        for(int j = m; j >= 1; --j) {
            while(!q.empty() && mp[i][j] < q.front().first) q.pop_front();
            q.push_front({mp[i][j], j});
            dp[i][j] = q.back().first;
            while(!q.empty() && q.back().second >= j + b - 1) q.pop_back();
        }
    }
    for(int i = 1; i <= n; ++i) {
        for(int j = 1; j <= m; ++j) {
            mp[i][j] = dp[i][j];
        }
    }
    for(int j = 1; j <= m; ++j) {
        while(!q.empty()) q.pop_back();
        for(int i = n; i >= 1; --i) {
            while(!q.empty() && mp[i][j] < q.front().first) q.pop_front();
            q.push_front({mp[i][j], i});
            dp[i][j] = q.back().first;
            while(!q.empty() && q.back().second >= i + a - 1) q.pop_back();
        }
    }
    LL ans = 0;
    for(int i = 1; i <= n - a + 1; ++i) {
        for(int j = 1; j <= m - b + 1; ++j) {
            ans += dp[i][j];
        }
    }
    printf("%lld\n", ans);
    return 0;
}
